Sub StockAnalysis():

    ' Setting intial variables
    Dim i As Integer
    Dim j As Integer
    Dim totalVolume As Double
        
    j = 1
    totalVolume = 0
    
    ' Gather data for ticker total volume
    For i = 2 To 526
        
       If Cells(i + 1, 1).Value = Cells(i, 1).Value Then
            totalVolume = totalVolume + Cells(i, 7)
       Else
            totalVolume = totalVolume + Cells(i, 7)
            
            ' Add summary table header when working across multiple worksheets
            If j = 1 Then
                Cells(j, 9).Value = "Ticker"
                Cells(j, 10).Value = "Yearly Change"
                Cells(j, 11).Value = "Percent Change"
                Cells(j, 12).Value = "Total Stock Volume"
            End If
            
            ' Post ticker and total volume to summary table
            j = j + 1
            Cells(j, 9).Value = Cells(i, 1).Value
            Cells(j, 12).Value = totalVolume
            
            ' Reset total volume
            totalVolume = 0
       End If
    
    Next i

End Sub
